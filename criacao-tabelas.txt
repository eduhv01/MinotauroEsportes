CREATE TABLE mensagem (
	id INTEGER PRIMARY KEY AUTO_INCREMENT,
	nome VARCHAR(100),
	email VARCHAR(100),
	mensagem TEXT NOT NULL
)

CREATE TABLE produto (
	id INTEGER PRIMARY KEY AUTO_INCREMENT,
	nome VARCHAR(100) NOT NULL,
	preco VARCHAR(50),
	imagem VARCHAR(40),
 	destaque BOOLEAN
)

CREATE TABLE usuario (
	id INTEGER PRIMARY KEY AUTO_INCREMENT,
	usuario VARCHAR(50) NOT NULL,
	senha VARCHAR(20) NOT NULL
)
